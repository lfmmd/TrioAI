MC_Halt

Commands a controlled motion stop.

Description:

MC_Halt is used to stop the axis under normal operation conditions. It is possible to set another motion command during deceleration of the axis, which will abort the MC_Halt and be executed immediately.

MC_Halt commands a controlled motion stop. The axis is moved to the state ‘Discrete Motion’, until the velocity is zero. With the ‘Done’ output set, the state is transferred to ‘Standstill’.

MC_Halt can be aborted by another motion command. If another function block is called while MC_Halt is busy, MC_Halt is aborted and the ‘CommandAborted’ output is set to TRUE. Therefore, MC_Halt does not guarantee the axis will go to zero velocity. For example, if MC_MoveVelocity is called while MC_Halt is ‘Busy’, MC_Halt will be aborted, and MC_MoveVelocity will be active.

 

There are three situations displayed in the above timing diagram:

  1. Normal application: the motion will be stopped by MC_Halt because there is no buffered motion command and no motion commands to abort halt process. The axis will change to ‘Standstill’.
  2. Aborted: the output ‘CommandAborted’ will be set if the motion command aborts MC_Halt process.
  3. Error: The new stopping speed will be used to stop the axis when it gets an axis error.

Make sure the input and output parameters are of data type indicated in the table below.

When MC_Halt is active, the next command can be issued. For example, a driverless vehicle detects an obstacle and needs to stop. MC_Halt is issued. Before the ‘Standstill’ is reached the obstacle is removed and the motion can be continued by setting another motion command, so the vehicle does not stop.

Description

OUTPUTS

BOOL

Error identification

Examples:

Example 1:

MC_MoveVelocity starts velocity control for axis0. When axis0 reaches the target speed and output ‘InVelocity’ is set, MC_Halt starts to decelerate axis0. Before speed of axis0 decreases to 0, MC_MoveVelocity aborts execution of MC_Halt and control axis0 again.

 

  Example

(* Global defines *)

(* Defines for MC_Halt example *)
#define STATE_READY   0
#define STATE_MOVE    4
#define STATE_HALT    5
#define STATE_MOVE_AGAIN 6
#define ERROR_STEP    7
#define STATE_ERROR   8
#define ERROR_NEXT    9

 

(* Variables define *)

VAR
 DirectionMode : MC_DIRECTION ;
  (*$desc=Motion direction*)
 BufferMode : MC_BUFFER_MODE ;
  (*$desc=Motion buffer mode setting variable*)
 State : USINT ;
  (*$desc=Current state*)
 bStart : BOOL ;
  (*$desc=Starting movement which is bound to IO5*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=3)*)
 Inst_MC_MoveVelocity : MC_MoveVelocity ;
 bHalt : BOOL ;
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=5)*)
 Inst_MC_Halt : MC_Halt ;
 Axis0Ref : lib:AXIS_REF ;
  (*$desc=Axis 0 reference*)
 ErrorID0 : UINT ;
  (*$desc=Error ID*)
 Inst_MC_ReadActualVelocity : MC_ReadActualVelocity ;
END_VAR

 

(* Variables initial *)  
DirectionMode := mcPositiveDirection;
BufferMode    := mcAborting;
State:= STATE_READY;
Axis0Ref.AxisNo := 0 ;

(* cyclic program *)
CASE State OF
     (* ...power and home axis before moving... *)
     (********************STATE_READY*********************)
    STATE_READY:
     (*wait for user to start movement*)
     IF (bStart) THEN         
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) ,  
                                Axis := Axis0Ref (*AXIS_REF*) ,
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) );
        
        Inst_MC_MoveVelocity(  Execute := TRUE (*BOOL*) ,
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) );
        
        State:= STATE_MOVE; 
     END_IF ;
     (********STATE MOVE*****************************************)
    STATE_MOVE: (*start moving*)     
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*) , 
                            Axis := Axis0Ref (*AXIS_REF*) , 
                            ContinuousUpdate := TRUE (*BOOL*) , 
                            Velocity := LREAL#1000.0 (*LREAL*) , 
                            Acceleration := LREAL#1000.0 (*LREAL*) , 
                            Deceleration := LREAL#1000.0 (*LREAL*) , 
                            Jerk := LREAL#500.0 (*LREAL*) , 
                            Direction := DirectionMode (*MC_DIRECTION*) , 
                            BufferMode := BufferMode (*MC_BUFFER_MODE*) );
    
     IF ((bHalt = TRUE ) AND (Inst_MC_MoveVelocity.InVelocity = TRUE )) THEN         
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) );
        Inst_MC_Halt( Execute := FALSE (*BOOL*) , 
                        Axis := Axis0Ref (*AXIS_REF*) , 
                        Deceleration := LREAL#2000.0 (*LREAL*) , 
                        Jerk := LREAL#1000.0 (*LREAL*) );
        
        State:= STATE_HALT;
     ELSIF (Inst_MC_MoveVelocity.Error) THEN
         (*Store ErrorID in a variable*)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        State:=ERROR_STEP;
     END_IF ;
     (********************STATE_HALT**********************)
    STATE_HALT:   (*halt movement*)
    Inst_MC_Halt( Execute := TRUE (*BOOL*) , 
                    Axis := Axis0Ref (*AXIS_REF*) , 
                    Deceleration := LREAL#2000.0 (*LREAL*) , 
                    Jerk := LREAL#1000.0 (*LREAL*) );
    Inst_MC_ReadActualVelocity( Axis := Axis0Ref (*AXIS_REF*) , 
                    Enable := TRUE (*BOOL*) );
    
     IF ((Inst_MC_ReadActualVelocity.Valid = TRUE ) AND  
        (Inst_MC_ReadActualVelocity.Velocity < 500 )) THEN
        Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        State:=STATE_MOVE_AGAIN;
     ELSIF (Inst_MC_Halt.Error = TRUE ) THEN
         (*Store ErrorID in a variable,*)
        ErrorID0:= Inst_MC_Halt.ErrorID;
        Inst_MC_ReadActualVelocity(  Enable := FALSE (*BOOL*) ,
                                        Axis := Axis0Ref (*AXIS_REF*) );
        Inst_MC_Halt( Execute := FALSE (*BOOL*) , 
                        Axis := Axis0Ref (*AXIS_REF*) , 
                        Deceleration := LREAL#2000.0 (*LREAL*) , 
                        Jerk := LREAL#1000.0 (*LREAL*) );
        State:=ERROR_STEP;
     ELSIF (Inst_MC_ReadActualVelocity.Error = TRUE ) THEN
         (*Store ErrorID in a variable,*)
        ErrorID0:= Inst_MC_ReadActualVelocity.ErrorID;
        Inst_MC_ReadActualVelocity( Enable := FALSE (*BOOL*) , 
                                         Axis := Axis0Ref (*AXIS_REF*) );
        Inst_MC_Halt( Execute := FALSE (*BOOL*) ,
                        Axis := Axis0Ref (*AXIS_REF*) ,  
                        Deceleration := LREAL#2000.0 (*LREAL*) , 
                        Jerk := LREAL#1000.0 (*LREAL*) );
        State:=ERROR_STEP;
     END_IF ;
     (********************STATE_MOVE_AGAIN**********************)
    STATE_MOVE_AGAIN: (*restart moving*)
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*) , 
                            Axis := Axis0Ref (*AXIS_REF*) , 
                            ContinuousUpdate := TRUE (*BOOL*) , 
                            Velocity := LREAL#1000.0 (*LREAL*) , 
                            Acceleration := LREAL#1000.0 (*LREAL*) , 
                            Deceleration := LREAL#1000.0 (*LREAL*) , 
                            Jerk := LREAL#500.0 (*LREAL*) , 
                            Direction := DirectionMode (*MC_DIRECTION*) , 
                            BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
     IF (Inst_MC_MoveVelocity.InVelocity = TRUE ) THEN
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        State := ERROR_NEXT;
     ELSIF (Inst_MC_MoveVelocity.Error) THEN
         (*Store ErrorID in a variable*)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) );  
                                State := ERROR_STEP;
     END_IF ;
        
    ERROR_STEP:
     (* Adding code here, when motion got errors *)
    
    NEXT_STEP:
     (* Adding code here, after MC_Stop calling *)
 
     END_CASE ;
