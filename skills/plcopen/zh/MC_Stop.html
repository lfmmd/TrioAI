MC_Stop

命令一个受控制的运动停止并将轴状态转换为“Stopping”。

描述

这个功能块命令轴停止并将轴状态转换为“Stopping”。它中止任何正在执行运动的功能块。

当轴处于“Stopping”状态时，其他FB无法在同一轴上进行任何运动。在轴达到“Velocity”零后，“Done”输出立即设置为TRUE。

只要“Execute”仍然为TRUE或“Velocity”尚未达到零，轴就保持在“Stoping”状态。在“Done”被设置后且“Execute”为FALSE后，轴将进入“Standstill”状态。

如果轴处于“ErrorStop”或“Disabled”状态，则无法执行此功能块。执行将产生一个错误。

在“Execute”更改为TRUE后，MC_Stop开始根据输入参数停止轴。速度的命令值是在当前速度的基础上创建的，“Deceleration”是减速速率的上限。减速度的命令值是根据当前的减速度的 “Jerk”值创建的。如果FB的任何输入参数设置为0，则将应用减速速率的上限值。

  • 指定减速和加加速度

速度的命令值是在JERK上限下以减速速率DECEL创建的。

如果“Deceleration”=0和“Jerk” =0同时成立，则在当前速度上立即停止，并在JERK上限下以DECEL上限执行减速过程。

确保输入和输出参数为下表所示的数据类型。

此FB主要用于紧急停止功能或异常情况

只要“Execute”高电平，轴保持在“停止”状态，可能不会执行任何其他运动命令。

如果“Deceleration”= 0，则功能块实现指定的行为。

名称数据类型描述
输入

输出

例
(*定义MC_Stop的示例*)
#define STATE_READY  0
#define STATE_MOVE   4
#define STATE_STOP   5
#define ERROR_STEP   7
#define STATE_ERROR  8
(*变量定义*)
VAR
State: USINT;
(*$desc=当前状态*)
BufferMode: MC_BUFFER_MODE;
(*$desc=运动缓冲模式设置变量*)
DirectionMode: MC_DIRECTION;
(*$desc=运动方向*)
bStart: BOOL;
(*$desc=绑定到IO3开始运动*)
 (*$profile=IO*)
 (*$embed=(VERS=1    MODE=0    POINT=3)*)
 Inst_MC_MoveVelocity : MC_MoveVelocity ;
 Axis0Ref : lib:AXIS_REF ;
(*$desc=轴0的轴参照*)
Inst_MC_Stop: MC_Stop;
bStop: BOOL;
(*$desc=绑定到IO4的停止开关*)
 (*$profile=IO*)
 (*$embed=(VERS=1    MODE=0    POINT=4)*)
 ErrorID0 : UINT ;
(*$desc=存储错误Id*)
END_VAR
(*变量初始化*)
DirectionMode := mcPositiveDirection;
BufferMode    := mcAborting;
State := STATE_READY;
Axis0Ref.AxisNo := 0;

(*循环程序*)
CASE State OF
    (* .. 移动前的上电并归零轴... *)
    (********************STATE_READY*********************)
    STATE_READY:
    (*等待用户开始移动*)
    IF (bStart) THEN
        Inst_MC_MoveVelocity( Execute := TRUE(*BOOL*),  
                                Axis := Axis0Ref(*AXIS_REF*),
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        State:= STATE_MOVE; 
    ELSE
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );        
    END_IF;
    
    (********STATE MOVE*****************************************)
    STATE_MOVE: (*开始运动*)
    IF ((bStop = TRUE) AND (Inst_MC_MoveVelocity.InVelocity = TRUE)) THEN
        
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*),
                                Axis := Axis0Ref(*AXIS_REF*),  
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        Inst_MC_Stop( Execute := TRUE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        
        State:= STATE_STOP;
    ELSIF (Inst_MC_MoveVelocity.Error) THEN
        (*在变量中储存错误ID *)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*),  
                            Axis := Axis0Ref(*AXIS_REF*),
                            ContinuousUpdate := TRUE(*BOOL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#500.0(*LREAL*), 
                            Direction := DirectionMode(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        State := ERROR_STEP;
    END_IF;
    (********************STATE_STOP**********************)
    STATE_STOP:  (*停止运动*)
    IF (Inst_MC_Stop.Done = TRUE) THEN
        Inst_MC_Stop( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        State := NEXT_STEP;
    ELSIF (Inst_MC_Stop.Error = TRUE) THEN
        (*在变量中储存错误ID *)
        ErrorID0:= Inst_MC_Stop.ErrorID;
        Inst_MC_Stop( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        State:=ERROR_STEP;
    END_IF;
    ERROR_STEP:
    (*当运动发生错误时，在这里添加代码*)
    
    NEXT_STEP:
    (* 在MC_Stop 调用后，在这里添加代码*)

    END_CASE;
