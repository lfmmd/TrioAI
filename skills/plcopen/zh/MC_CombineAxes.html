MC_CombineAxes

将两个轴的运动组合到第三个轴，组合的方法可以选择。

描述

MC_CombineAxes将两个轴的运动组合到第三个轴，组合的方法可以选择。本质上，它是基于输入轴的两个位置设定值计算一个新的位置设定值。这个FB反映在状态图中，就像一个同步运动类型。

MC_CombineAxes支持两种组合模式:mcAddAxes和mcSubAxes。

  • mcAddAxes:从轴位置= 主轴1位置(相对位置)+ 主轴2位置(相对位置)。
  • mcSubAxes:从轴位置= 主轴1位置(相对位置)- 主轴2位置(相对位置)。

MC_CombineAxes不受'Master1 '或'Master2 '错误的影响。当错误被清除并且主轴恢复运动后，从轴将恢复组合定位操作。主轴不受从轴错误的影响。

作为应用示例，可以使用与移动皮带上的对象同步的单独轮廓，或具有灵活覆盖距离的旋转刀来切割。

确保输入和输出参数为下表所示的数据类型。

要停止运动，MC_CombineAxes必须被另一个FB发出的新命令打断。

例1

轴1是MC_CamIn的主轴。轴0是从轴。主轴将沿正弦波速度轮廓定义的凸轮table移动。当轴1与主轴同步时，它将作为轴0以MC_GearIn定义的方式移动。当轴1与轴0同步时，轴3将以MC_MoveAbsolute开始。轴0和轴1以 mcAddAxes模式组合到轴3。下图显示了这些轴的速度轮廓。

(*定义MC_CombineAxes的例子*)
#define STATE_READY     0
#define STATE_CAMIN     1
#define STATE_GEARIN    2
#define STATE_COMBINE   3
#define STATE_ERROR     8
(*定义MC_CombineAxes中使用的变量*)
VAR
 state : USINT ;
(*$desc=当前状态*)
Inst_MC_CamIn: MC_CamIn;
(*$desc=MC_CamIn 实例*)
Axis0Ref: lib:AXIS_REF;
(*$desc=参考轴0*)
Axis2Ref: lib:AXIS_REF;
(*$desc=参考轴2 *)
CamId: UINT;
(*$desc=Cam表ID*)
bGearStart: BOOL:= FALSE;
(*$desc=启动齿轮运动，它被绑定到IO6*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=7)*)
Inst_MC_GearIn:MC_GearIn;
Axis1Ref: lib:AXIS_REF;
(*$desc=从轴，将在齿轮运动时移动*)
bStartCombine: BOOL值;
(*$desc=从轴，将在联合运动时移动*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=7)*)
 Inst_MC_CombineAxes : MC_CombineAxes ;
 Axis3Ref : lib:AXIS_REF ;
 bStop : BOOL := FALSE ;
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=1)*)
 Inst_MC_Stop1 : MC_Stop ;
 Inst_MC_Stop2 : MC_Stop ;
 Inst_MC_Stop3 : MC_Stop ;
 bStart : BOOL := FALSE ;
(*$desc=启动联合运动示例*)
(*$profile=IO*)
(*$embed=(VERS=1    MODE=0    POINT=0)*)
 SrcValue : MC_SOURCE := mcSetValue ;
 StartMode : MC_START_MODE ;
 CombMode : MC_COMBINE_MODE ;
 Inst_MC_MoveAbsolute : MC_MoveAbsolute ;
 mDirection : MC_DIRECTION ;
 BufferMode : MC_BUFFER_MODE ;
END_VAR
(*变量初始化*)
CamId:= UINT #25616;(*这个Cam表应该在使用之前创建*)
Axis0Ref.AxisNo:= 0;(*凸轮运动的从轴*)
Axis1Ref.AxisNo:= 1;(*齿轮运动的从轴*) ·
Axis2Ref.AxisNo:= 2;(*凸轮运动的主轴*)
Axis3Ref.AxisNo:= 3;(*联合运动的从轴*)
SrcValue := mcSetValue;
StartMode := mcAbsolute;
CombMode := mcAddAxes;
mDirection := mcPositiveDirection;
BufferMode := mcAborting;


(*循环程序*)
CASE state OF
    (* ... 移动前的电源和主轴... *)
    (********************STATE_READY*********************)
    STATE_READY:
   (*等待用户开始移动*)
    IF (bStart) THEN
        Inst_MC_CamIn( Execute := FALSE (*BOOL*),
                        Master := Axis2Ref(*AXIS_REF*), 
                        Slave := Axis0Ref (*AXIS_REF*),         
                        MasterOffset := LREAL#0.0 (*LREAL*), 
                        SlaveOffset := LREAL#0.0 (*LREAL*), 
                        MasterScaling := REAL#1.0 (*REAL*), 
                        SlaveScaling := REAL#1.0 (*REAL*), 
                        MasterStartDistance := LREAL#0.0 (*LREAL*), 
                        MasterSyncPosition := LREAL#0.0 (*LREAL*), 
                        StartMode := StartMode(*MC_START_MODE*), 
                        MasterValueSource := SrcValue(*MC_SOURCE*), 
                        CamTableID := CamId(*UINT*) );
        
        state:= STATE_CAMIN;
    END_IF;
    (********STATE CAM IN*****************************************)
    STATE_CAMIN:  (*start cam in*)
    Inst_MC_CamIn(Execute := TRUE (*BOOL*), 
                    Master := Axis2Ref(*AXIS_REF*), 
                    Slave := Axis0Ref (*AXIS_REF*), 
                    MasterOffset := LREAL#0.0 (*LREAL*), 
                    SlaveOffset := LREAL#0.0 (*LREAL*), 
                    MasterScaling := REAL#1.0 (*REAL*), 
                    SlaveScaling := REAL#1.0 (*REAL*), 
                    MasterStartDistance := LREAL#0.0 (*LREAL*), 
                    MasterSyncPosition := LREAL#0.0 (*LREAL*), 
                    StartMode := StartMode(*MC_START_MODE*), 
                    MasterValueSource := SrcValue(*MC_SOURCE*), 
                    CamTableID := CamId(*UINT*) );
    
    IF ( (bGearStart =TRUE) AND (Inst_MC_CamIn.InSync)) THEN
        Inst_MC_GearIn( Execute := FALSE(*BOOL*), 
                        Master  := Axis2Ref(*AXIS_REF*), 
                        Slave  := Axis1Ref(*AXIS_REF*),         
                        ContinuousUpdate := TRUE(*BOOL*), 
                        RatioNumerator := INT#1(*INT*), 
                        RatioDenominator := UINT#1(*UINT*), 
                        MasterValueSource := SrcValue(*MC_SOURCE*), 
                        Acceleration := LREAL#1000.0(*LREAL*), 
                        Deceleration := LREAL#1000.0(*LREAL*), 
                        Jerk := LREAL#100.0(*LREAL*) );
          
        state:=STATE_GEARIN;
    ELSIF (Inst_MC_CamIn.Error) THEN        
        Inst_MC_CamIn( Execute := FALSE (*BOOL*), 
                        Master := Axis2Ref(*AXIS_REF*), 
                        Slave := Axis0Ref (*AXIS_REF*),         
                        MasterOffset := LREAL#0.0 (*LREAL*), 
                        SlaveOffset := LREAL#0.0 (*LREAL*), 
                        MasterScaling := REAL#1.0 (*REAL*), 
                        SlaveScaling := REAL#1.0 (*REAL*), 
                        MasterStartDistance := LREAL#0.0 (*LREAL*), 
                        MasterSyncPosition := LREAL#0.0 (*LREAL*), 
                        StartMode := StartMode(*MC_START_MODE*), 
                        MasterValueSource := SrcValue(*MC_SOURCE*), 
                        CamTableID := CamId(*UINT*) );
        
        state:=STATE_ERROR;
    END_IF;
    (********STATE GEAR IN*****************************************)
    STATE_GEARIN:  (*start cam in*)
    
    Inst_MC_GearIn( Execute := TRUE(*BOOL*), 
                    Master  := Axis2Ref(*AXIS_REF*), 
                    Slave  := Axis1Ref(*AXIS_REF*),     
                    ContinuousUpdate := TRUE(*BOOL*), 
                    RatioNumerator := INT#1(*INT*), 
                    RatioDenominator := UINT#1(*UINT*), 
                    MasterValueSource := SrcValue(*MC_SOURCE*), 
                    Acceleration := LREAL#1000.0(*LREAL*), 
                    Deceleration := LREAL#1000.0(*LREAL*), 
                    Jerk := LREAL#100.0(*LREAL*) );
    
    IF ((bStartCombine =TRUE) AND (Inst_MC_GearIn.InGear)) THEN
        Inst_MC_CombineAxes( Execute := FALSE(*BOOL*),
                            Master1 := Axis0Ref(*AXIS_REF*),
                            Master2 := Axis1Ref(*AXIS_REF*), 
                            Slave := Axis3Ref(*AXIS_REF*),                              
                            ContinuousUpdate := TRUE(*BOOL*), 
                            CombineMode := CombMode(*MC_COMBINE_MODE*),
                            GearRatioNumeratorM1 := INT#1(*INT*), 
                            GearRatioDenominatorM1 := INT#1(*INT*), 
                            GearRatioNumeratorM2 := INT#1(*INT*), 
                            GearRatioDenominatorM2 := INT#1(*INT*), 
                            MasterValueSourceM1 := SrcValue(*MC_SOURCE*), 
                            MasterValueSourceM2 := SrcValue(*MC_SOURCE*) );
        
        Inst_MC_MoveAbsolute( Execute := FALSE(*BOOL*),
                            Axis := Axis3Ref(*AXIS_REF*),
                            ContinuousUpdate := FALSE(*BOOL*), 
                            Position := LREAL#10000.0(*LREAL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#1000.0(*LREAL*), 
                            Direction := mDirection(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        state:=STATE_COMBINE;
    ELSIF (Inst_MC_GearIn.Error) THEN        
        Inst_MC_GearIn( Execute := FALSE (*BOOL*), 
                            Master  := Axis2Ref(*AXIS_REF*), 
                            Slave  := Axis1Ref(*AXIS_REF*),
                            ContinuousUpdate := TRUE(*BOOL*), 
                            RatioNumerator := INT#1(*INT*), 
                            RatioDenominator := UINT#1(*UINT*), 
                            MasterValueSource := SrcValue(*MC_SOURCE*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#100.0(*LREAL*) );
        state:=STATE_ERROR;
    END_IF;
    (********STATE COMBINE*****************************************)
    STATE_COMBINE:  (*start combine *)    
    Inst_MC_CombineAxes( Execute := TRUE(*BOOL*),
                            Master1 := Axis0Ref(*AXIS_REF*),
                            Master2 := Axis1Ref(*AXIS_REF*),  
                            Slave := Axis3Ref(*AXIS_REF*),
                            ContinuousUpdate := TRUE(*BOOL*), 
                            CombineMode := CombMode(*MC_COMBINE_MODE*),
                            GearRatioNumeratorM1 := INT#1(*INT*), 
                            GearRatioDenominatorM1 := INT#1(*INT*), 
                            GearRatioNumeratorM2 := INT#1(*INT*), 
                            GearRatioDenominatorM2 := INT#1(*INT*), 
                            MasterValueSourceM1 := SrcValue(*MC_SOURCE*), 
                            MasterValueSourceM2 := SrcValue(*MC_SOURCE*) );
    
    Inst_MC_MoveAbsolute( Execute := TRUE(*BOOL*),
                            Axis := Axis3Ref(*AXIS_REF*), 
                            ContinuousUpdate := FALSE(*BOOL*), 
                            Position := LREAL#10000.0(*LREAL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#1000.0(*LREAL*), 
                            Direction := mDirection(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) );
    
    IF ( bStop = TRUE) THEN
        Inst_MC_Stop1( Execute := TRUE(*BOOL*), 
                        Axis := Axis3Ref(*AXIS_REF*),
                        Deceleration := LREAL#1000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        
        Inst_MC_Stop2( Execute := TRUE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*),
                        Deceleration := LREAL#1000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        
        Inst_MC_Stop3( Execute := TRUE(*BOOL*), 
                        Axis := Axis1Ref(*AXIS_REF*),
                        Deceleration := LREAL#1000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        
        state:=STATE_READY;
    ELSIF (Inst_MC_CombineAxes.Error) THEN
        
        Inst_MC_CombineAxes( Execute := FALSE(*BOOL*),
                                Master1 := Axis0Ref(*AXIS_REF*),
                                Master2 := Axis1Ref(*AXIS_REF*), 
                                Slave := Axis3Ref(*AXIS_REF*),                                  
                                ContinuousUpdate := TRUE(*BOOL*), 
                                CombineMode := CombMode(*MC_COMBINE_MODE*),
                                GearRatioNumeratorM1 := INT#1(*INT*), 
                                GearRatioDenominatorM1 := INT#1(*INT*), 
                                GearRatioNumeratorM2 := INT#1(*INT*), 
                                GearRatioDenominatorM2 := INT#1(*INT*), 
                                MasterValueSourceM1 := SrcValue(*MC_SOURCE*), 
                                MasterValueSourceM2 := SrcValue(*MC_SOURCE*) );
        
        state:=STATE_ERROR;
    END_IF;
    (********STATE ERROR*****************************************)
    STATE_ERROR:  
    (* 在此添加用户代码*)

END_CASE;
