MC_Halt

指定一个受控制的运动停止。

描述

MC_Halt用于在正常运行条件下停止轴。可以在轴减速期间设置另一个运动命令，这将中止MC_Halt并立即执行新的运动命令。

MC_Halt命令一个受控的运动停止。轴运动转移到“Discrete Motion”状态，直到速度为零。设置了“Done”输出后，状态转移到“Standstill”。

MC_Halt可以被另一个运动命令终止。如果在MC_Halt繁忙时调用另一个函数块，则MC_Halt被中止，并且“CommandAborted”输出设置为TRUE。因此，MC_Halt不能保证轴的速度为零。例如，如果在MC_Halt处于“Busy”状态时调用MC_MoveVelocity，MC_Halt将被中止，而MC_MoveVelocity将处于活动状态。

以上时序图中显示的情况有三种:

  • 正常应用：运动将被MC_Halt停止，因为没有缓冲的运动命令，也没有运动命令来中止暂停进程。轴将变为“Standstill”。
  • 终止：如果运动命令终止MC_Halt进程，输出“CommandAborted”将被设置。
  • 错误：当获得轴错误时，将使用新的停止速度来停止轴。

确保输入和输出参数为下表所示的数据类型。

当MC_Halt处于活动状态时，可以发出下一个命令。例如，无人驾驶车辆检测到障碍物，需要停车。发出MC_Halt。在到达“Standstill”之前，障碍物被移除，可以通过设置另一个运动命令继续运动，因此车辆不会停止。

输入/输出

名称数据类型描述
输入
ExecuteBOOL在上升沿开始动作
AxisAXIS_REF参考轴
DecelerationLREAL“减速度”值[u/]
JerkLREAL“加加速度”值[u/]
输出
DoneBOOL达到零速度
BusyBOOLFB尚未完成，预计会有新的输出值。
CommandAbortedBOOL“Command”被另一个命令终止。
ErrorBOOL表示FB中发生了错误。
ErrorIDUINT错误标识

例1

MC_MoveVelocity启动轴0的速度控制。当轴0达到目标速度并设置输出“InVelocity”时，MC_Halt开始为轴0减速。在轴0的速度降低到0之前，MC_MoveVelocity中止MC_Halt的执行并再次控制轴0。

例
(* 全局定义 *)
(*MC_Halt示例定义 *)
#define STATE_READY  0
#define STATE_MOVE   4
#define STATE_HALT   5
#define STATE_MOVE_AGAIN 6
#define ERROR_STEP   7
#define STATE_ERROR  8
#define ERROR_NEXT   9
(*变量定义*)
VAR
DirectionMode: MC_DIRECTION;
(*$desc=运动方向*)
BufferMode: MC_BUFFER_MODE;
(*$desc=运动缓冲模式设置变量*)
State: USINT;
(*$desc=当前状态*)
bStart: BOOL;
(*$desc=绑定到IO5的起始运动*)
(*$profile=IO*)
(*$embed=(VERS=1 MODE=0 POINT=3)*)
Inst_MC_MoveVelocity: MC_MoveVelocity;
bHalt: BOOL值;
(*$profile=IO*)
(*$embed=(VERS=1 MODE=0 POINT=5)*)
Inst_MC_Halt: MC_Halt;
Axis0Ref: lib:AXIS_REF;
(*$desc=Axis 0 参照*)
ErrorID0: UINT;
(*$desc=Error ID*)
Inst_MC_ReadActualVelocity: MC_ReadActualVelocity;
END_VAR
(*变量初始化*)
DirectionMode := mcPositiveDirection;
BufferMode    := mcAborting;
State:= STATE_READY;
Axis0Ref.AxisNo := 0;

(*循环程序*)
CASE State OF
    (* ... 移动前的电源和主轴... *)
    (********************STATE_READY*********************)
    STATE_READY:
    (*等待用户开始移动*)
    IF (bStart) THEN        
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*),  
                                Axis := Axis0Ref(*AXIS_REF*),
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        Inst_MC_MoveVelocity(  Execute := TRUE (*BOOL*),
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        State:= STATE_MOVE; 
    END_IF;
    (********STATE MOVE*****************************************)
    STATE_MOVE: (*开始移动*)    
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*), 
                            Axis := Axis0Ref(*AXIS_REF*), 
                            ContinuousUpdate := TRUE(*BOOL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#500.0(*LREAL*), 
                            Direction := DirectionMode(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) );
    
    IF ((bHalt = TRUE) AND (Inst_MC_MoveVelocity.InVelocity = TRUE)) THEN        
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        Inst_MC_Halt( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        
        State:= STATE_HALT;
    ELSIF (Inst_MC_MoveVelocity.Error) THEN
        (*在变量中储存错误ID*)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        State:=ERROR_STEP;
    END_IF;
    (********************STATE_HALT**********************)
    STATE_HALT:  (*停止移动*)
    Inst_MC_Halt( Execute := TRUE(*BOOL*), 
                    Axis := Axis0Ref(*AXIS_REF*), 
                    Deceleration := LREAL#2000.0(*LREAL*), 
                    Jerk := LREAL#1000.0(*LREAL*) );
    Inst_MC_ReadActualVelocity( Axis := Axis0Ref(*AXIS_REF*), 
                    Enable := TRUE(*BOOL*) );
    
    IF((Inst_MC_ReadActualVelocity.Valid = TRUE) AND 
        (Inst_MC_ReadActualVelocity.Velocity < 500))THEN
        Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        State:=STATE_MOVE_AGAIN;
    ELSIF (Inst_MC_Halt.Error = TRUE) THEN
        (*在变量中储存错误ID *)
        ErrorID0:= Inst_MC_Halt.ErrorID;
        Inst_MC_ReadActualVelocity(  Enable := FALSE(*BOOL*),
                                        Axis := Axis0Ref(*AXIS_REF*) );
        Inst_MC_Halt( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        State:=ERROR_STEP;
    ELSIF (Inst_MC_ReadActualVelocity.Error = TRUE) THEN
        (*在变量中储存错误ID *)
        ErrorID0:= Inst_MC_ReadActualVelocity.ErrorID;
        Inst_MC_ReadActualVelocity( Enable := FALSE(*BOOL*), 
                                         Axis := Axis0Ref(*AXIS_REF*));
        Inst_MC_Halt( Execute := FALSE(*BOOL*),
                        Axis := Axis0Ref(*AXIS_REF*),  
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk := LREAL#1000.0(*LREAL*) );
        State:=ERROR_STEP;
    END_IF;
    (********************STATE_MOVE_AGAIN**********************)
    STATE_MOVE_AGAIN: (*重新开始移动*)
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*), 
                            Axis := Axis0Ref(*AXIS_REF*), 
                            ContinuousUpdate := TRUE(*BOOL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#500.0(*LREAL*), 
                            Direction := DirectionMode(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
    IF(Inst_MC_MoveVelocity.InVelocity = TRUE) THEN
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        State := ERROR_NEXT;
    ELSIF(Inst_MC_MoveVelocity.Error) THEN
        (*在变量中储存错误ID *)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );  
                                State := ERROR_STEP;
    END_IF;
        
    ERROR_STEP:
    (* 当运动出错时，在此添加代码*)
    
    NEXT_STEP:
    (* 在调用MC_Stop后，在此添加代码*)
 
    END_CASE;
