MC_Reset

重置所有内部与轴相关的错误。

描述

MC_Reset通过重置所有内部与轴相关的错误，将状态从“ErrorStop”转换为“Standstill”或“Disabled”。它不会影响FB实例的输出。

如果轴上有驱动器相关的错误，应当首先清除驱动器上的错误，然后才可以执行复位操作。

请确保输入和输出参数为下表所示的数据类型。

MC_Reset可以被重新触发，但是当前一个进程没有完成重置操作时，重新触发该块将产生一个错误。

如果您在轴不处于“ErrorStop”状态时执行此FB，则会产生错误。

它用于将轴状态从“ErrorStop”更改为“Standstill”或“Disabled”，但不清除轴中的错误。在执行重置操作之前，需要先清除轴上的错误。

输入/输出

名称数据类型描述
输入
ExecuteBOOL在上升沿处写入参数的值
AxisAXIS_REF参考轴
输出
DoneBOOL达到零速度
BusyBOOLFB尚未完成，预计会有新的输出值。
ErrorBOOL表示FB中发生了错误。
ErrorIDUINT错误标识

例

例1

轴0出现一些错误，处于“ErrorStop”状态。如果所有错误都解决了，这些错误将被MC_Reset清除。然后，轴0由MC_Power供电。

例
(*定义MC_Reset的示例*)
#define STATE_READY  0
#define STATE_RESET  2
#define NEXT_STEP    3
#define STATE_MOVE   4
#define ERROR_STEP   7
(*变量定义*)
VAR
DirectionMode: MC_DIRECTION;
(*$desc=运动方向*)
BufferMode: MC_BUFFER_MODE;
(*$desc=用于设置的缓冲区模式变量*)
State: USINT;
(*$desc=当前状态*)
bReset: BOOL;
  (*$desc=Reset swich status which is bound to IO6*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=6)*)
 Inst_MC_Reset : MC_Reset ;
 Inst_MC_MoveVelocity : MC_MoveVelocity ;
 Axis0Ref : lib:AXIS_REF ;
  (*$desc=参考轴0*)
 ErrorID0 : UINT ;
  (*$desc=错误ID*)
END_VAR
(*变量初始化*)
DirectionMode:= mcPositiveDirection;
BufferMode:= mcAborting;
状态:= STATE_READY;
Axis0Ref。AxisNo:= 0;

(*循环程序*)
CASE State OF
    (********************STATE_READY*********************)
    STATE_READY:
    (*等待用户重置轴错误*)
    IF (bReset) THEN
        Inst_MC_Reset( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*) );
        Inst_MC_Reset( Execute := TRUE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*) );
        State:= STATE_RESET; 
    END_IF;
    (********STATE RESET*****************************************)
    STATE_RESET: (*重置轴错误*)
    IF ( Inst_MC_Reset.Done = TRUE ) THEN
        
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        
        Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        
        State:= STATE_MOVE;
    ELSIF (Inst_MC_Reset.Error) THEN
        (*在变量中存储错误ID*)
        ErrorID0:= Inst_MC_Reset.ErrorID;
        Inst_MC_Reset( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*) );
        State:= ERROR_STEP;
    END_IF;
    (********************STATE_MOVE**********************)
    STATE_MOVE:  (*停止移动*)
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*),  
                            Axis := Axis0Ref(*AXIS_REF*),
                            ContinuousUpdate := TRUE(*BOOL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#500.0(*LREAL*), 
                            Direction := DirectionMode(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
    IF(Inst_MC_MoveVelocity.InVelocity = TRUE) THEN
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*),
                                Axis := Axis0Ref(*AXIS_REF*),  
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        State := NEXT_STEP;
    ELSIF ( Inst_MC_MoveVelocity.Error) THEN
        (*在变量中存储错误ID *)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) ); 
        State := ERROR_STEP;
    END_IF;
    ERROR_STEP:
    (*在此添加代码，当运动出错时*)
    
    NEXT_STEP:
    (*在此添加代码，在MC_Stop调用后*)

    END_CASE;
