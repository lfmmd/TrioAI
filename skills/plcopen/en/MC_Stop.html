MC_Stop

Commands a controlled motion stop and transfers the axis to the state ‘Stopping’.

Description

This Function Block commands an axis to stop and transfers the axis into the state of ‘Stopping’. It aborts any ongoing Function Block executing motion.

When the axis is in the state of ‘Stopping’, no other FB can perform any motion on the same axis. After the axis has reached ‘Velocity’ zero, the ‘Done’ output is set to TRUE immediately.

The axis remains in the state of ‘Stopping’ as long as ‘Execute’ is still TRUE or ‘Velocity’ zero has not yet reached. After ‘Done’ is SET and ‘Execute’ is FALSE, the axis will go to the state ‘Standstill’.

If the axis is in the state of ‘ErrorStop’ or ‘Disabled’, this Function Block cannot be executed. Doing so will result in an error.

MC_Stop starts to stop an axis according to input parameters after ‘Execute’ changing to TRUE. The command value of the velocity is created based on the current velocity with ‘Deceleration’ as the upper limit of the deceleration rate. The command value of the deceleration is created based on the current deceleration with the value of ‘Jerk’. If any input parameter of FB is set 0, its maximum limitation values will be applied.

  • Specifying Deceleration and Jerk

The relationships between the deceleration and velocity when Jerk is set to 0 and any other value are shown below.

  1. Jerk is set to 0

    The command value for the velocity is created with deceleration rate DECEL at maximum limit JERK.

    Vt: Velocity when deceleration starts

    DECEL: Specified deceleration rate

    JERK: Specified Jerk.

  2. Jerk is set to any value other than 0

    The command value for the velocity is created based on the current velocity with DECEL as the upper limit to the deceleration rate.

If ‘Deceleration’ = 0 and ‘Jerk’=0 simultaneously, an immediate stop is executed on the current velocity with the maximum limit DECEL at maximum limit JERK.

Make sure the input and output parameters are of data type indicated in the table below.

This FB is primarily intended for emergency stop functionality or exception situations

As long as ‘Execute’ is high, the axis remains in the state ‘Stopping’ and may not be executing any other motion command.

If ‘Deceleration’ = 0, the behavior of the function block is implementation specified.

Examples

Example 1

Inst_MC_MoveVelocity starts velocity control for axis0. When ‘InVelocity’ of  Inst_MC_MoveVelocity is set, Inst_MC_Stop is executed and decelerates axis0 from current velocity to 0.

Example
(* Defines for MC_Stop example *)
#define STATE_READY  0
#define STATE_MOVE   4
#define STATE_STOP   5
#define ERROR_STEP   7
#define STATE_ERROR  8
(* Variables define *)
VAR
 State : USINT ;
  (*$desc=Current state*)
 BufferMode : MC_BUFFER_MODE ;
  (*$desc=Motion buffer mode setting variable*)
 DirectionMode : MC_DIRECTION ;
  (*$desc=Motion direction*)
 bStart : BOOL ;
  (*$desc=Bound to IO3 for starting motion*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=3)*)
 Inst_MC_MoveVelocity : MC_MoveVelocity ;
 Axis0Ref : lib:AXIS_REF ;
  (*$desc=Axis reference for axis 0*)
 Inst_MC_Stop : MC_Stop ;
 bStop : BOOL ;
  (*$desc=Stop switch which is bound to IO4*)
  (*$profile=IO*)
  (*$embed=(VERS=1    MODE=0    POINT=4)*)
 ErrorID0 : UINT ;
  (*$desc=Storing ErrorId*)
END_VAR
(* Variables initial *) 
DirectionMode := mcPositiveDirection;
BufferMode    := mcAborting;
State := STATE_READY;
Axis0Ref.AxisNo := 0;

(* cyclic program *)
CASE State OF
    (* ...power and home axis before moving... *)
    (********************STATE_READY*********************)
    STATE_READY:
    (*wait for user to start movement*)
    IF (bStart) THEN
        Inst_MC_MoveVelocity( Execute := TRUE(*BOOL*),  
                                Axis := Axis0Ref(*AXIS_REF*),
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        State:= STATE_MOVE; 
    ELSE
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*), 
                                Axis := Axis0Ref(*AXIS_REF*), 
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );        
    END_IF;
    
    (********STATE MOVE*****************************************)
    STATE_MOVE: (*start moving*)
    IF ((bStop = TRUE) AND (Inst_MC_MoveVelocity.InVelocity = TRUE)) THEN
        
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*),
                                Axis := Axis0Ref(*AXIS_REF*),  
                                ContinuousUpdate := TRUE(*BOOL*), 
                                Velocity := LREAL#1000.0(*LREAL*), 
                                Acceleration := LREAL#1000.0(*LREAL*), 
                                Deceleration := LREAL#1000.0(*LREAL*), 
                                Jerk := LREAL#500.0(*LREAL*), 
                                Direction := DirectionMode(*MC_DIRECTION*), 
                                BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        
        Inst_MC_Stop( Execute := TRUE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        
        State:= STATE_STOP;
    ELSIF (Inst_MC_MoveVelocity.Error) THEN
        (*Store ErrorID in a variable*)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE(*BOOL*),  
                            Axis := Axis0Ref(*AXIS_REF*),
                            ContinuousUpdate := TRUE(*BOOL*), 
                            Velocity := LREAL#1000.0(*LREAL*), 
                            Acceleration := LREAL#1000.0(*LREAL*), 
                            Deceleration := LREAL#1000.0(*LREAL*), 
                            Jerk := LREAL#500.0(*LREAL*), 
                            Direction := DirectionMode(*MC_DIRECTION*), 
                            BufferMode := BufferMode(*MC_BUFFER_MODE*) );
        State := ERROR_STEP;
    END_IF;
    (********************STATE_STOP**********************)
    STATE_STOP:  (*stop movement*)
    IF (Inst_MC_Stop.Done = TRUE) THEN
        Inst_MC_Stop( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        State := NEXT_STEP;
    ELSIF (Inst_MC_Stop.Error = TRUE) THEN
        (*Store ErrorID in a variable,*)
        ErrorID0:= Inst_MC_Stop.ErrorID;
        Inst_MC_Stop( Execute := FALSE(*BOOL*), 
                        Axis := Axis0Ref(*AXIS_REF*), 
                        Deceleration := LREAL#2000.0(*LREAL*), 
                        Jerk  := LREAL#1000.0(*LREAL*) );
        State:=ERROR_STEP;
    END_IF;
    ERROR_STEP:
    (* Adding code here, when motion got errors *)
    
    NEXT_STEP:
    (* Adding code here, after MC_Stop calling *)

    END_CASE;
