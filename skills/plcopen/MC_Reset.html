MC_Reset

Resets all internal axis-related errors.

Description:

MC_Reset makes the transition from the state ‘ErrorStop’ to ‘Standstill’ or ‘Disabled’ by resetting all internal axis-related errors. It does not affect the output of the FB instances.

If there are drive-related errors on the axis, the drive errors are to be cleared first then reset operation can be performed.

Make sure the input and output parameters are of data type indicated in the table below.

MC_Reset can be retriggered but when the previous process has not finished the reset operation, retriggering the block will generate an error.

It will generate an error if you execute this FB when the axis is not in ‘ErrorStop’.

It is used to change the axis state from ‘ErrorStop’ to ‘Standstill’ or ‘’Disabled’, not to clear the error in the axis. Axis’ errors needs to be cleared before executing reset operation.

Inputs/Outputs:

Name

Data Type

Description

INPUTS

Execute

BOOL

Write the value of the parameter at rising edge

Axis

AXIS_REF

Reference to the axis

OUTPUTS

Done

BOOL

Zero velocity reached

Busy

BOOL

The FB is not finished and a new output value is to be expected.

Error

BOOL

Signals that an error has occurred within the FB.

ErrorID

UINT

Error identification

Examples:

Example 1:

Axis0 has some errors and is in ‘ErrorStop’ state. Theses errors will be cleared by MC_Reset , if all errors have been resolved.  Axis0 is then powered by MC_Power.

 

 Example

(* Defines for MC_Reset example *)
#define STATE_READY   0
#define STATE_RESET   2
#define NEXT_STEP     3
#define STATE_MOVE    4
#define ERROR_STEP    7

(* Variable defines *)

VAR
 DirectionMode : MC_DIRECTION ;
   (*$desc=Motion direction*)
 BufferMode : MC_BUFFER_MODE ;
   (*$desc=Buffer mode variable for setting*)
 State : USINT ;
   (*$desc=Current state*)
 bReset : BOOL ;
   (*$desc=Reset swich status which is bound to IO6*)
   (*$profile=IO*)
   (*$embed=(VERS=1    MODE=0    POINT=6)*)
 Inst_MC_Reset : MC_Reset ;
 Inst_MC_MoveVelocity : MC_MoveVelocity ;
 Axis0Ref : lib:AXIS_REF ;
   (*$desc=Axis 0 reference*)
 ErrorID0 : UINT ;
   (*$desc=Error ID*)
END_VAR

(* Variables initial *)  
DirectionMode := mcPositiveDirection;
BufferMode    := mcAborting;
State := STATE_READY;
Axis0Ref.AxisNo := 0 ;

(* cyclic program *)
CASE State OF
     (********************STATE_READY*********************)
    STATE_READY:
     (*wait for user to reset axis'error*)
     IF (bReset) THEN
        Inst_MC_Reset( Execute := FALSE (*BOOL*) , 
                        Axis := Axis0Ref (*AXIS_REF*) );
        Inst_MC_Reset( Execute := TRUE (*BOOL*) , 
                        Axis := Axis0Ref (*AXIS_REF*) );
        State:= STATE_RESET; 
     END_IF ;
     (********STATE RESET*****************************************)
    STATE_RESET: (*reset axis's error*)
     IF ( Inst_MC_Reset.Done = TRUE ) THEN
        
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        
        Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        
        State:= STATE_MOVE;
     ELSIF (Inst_MC_Reset.Error) THEN
         (*Store ErrorID in a variable*)
        ErrorID0:= Inst_MC_Reset.ErrorID;
        Inst_MC_Reset( Execute := FALSE (*BOOL*) , 
                        Axis := Axis0Ref (*AXIS_REF*) );
        State:= ERROR_STEP;
     END_IF ;
     (********************STATE_MOVE**********************)
    STATE_MOVE:   (*halt movement*)
    Inst_MC_MoveVelocity( Execute := TRUE (*BOOL*) ,  
                            Axis := Axis0Ref (*AXIS_REF*) ,
                            ContinuousUpdate := TRUE (*BOOL*) , 
                            Velocity := LREAL#1000.0 (*LREAL*) , 
                            Acceleration := LREAL#1000.0 (*LREAL*) , 
                            Deceleration := LREAL#1000.0 (*LREAL*) , 
                            Jerk := LREAL#500.0 (*LREAL*) , 
                            Direction := DirectionMode (*MC_DIRECTION*) , 
                            BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
     IF (Inst_MC_MoveVelocity.InVelocity = TRUE ) THEN
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) ,
                                Axis := Axis0Ref (*AXIS_REF*) ,  
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        State := NEXT_STEP;
     ELSIF ( Inst_MC_MoveVelocity.Error) THEN
         (*Store ErrorID in a variable*)
        ErrorID0:= Inst_MC_MoveVelocity.ErrorID;
        Inst_MC_MoveVelocity( Execute := FALSE (*BOOL*) , 
                                Axis := Axis0Ref (*AXIS_REF*) , 
                                ContinuousUpdate := TRUE (*BOOL*) , 
                                Velocity := LREAL#1000.0 (*LREAL*) , 
                                Acceleration := LREAL#1000.0 (*LREAL*) , 
                                Deceleration := LREAL#1000.0 (*LREAL*) , 
                                Jerk := LREAL#500.0 (*LREAL*) , 
                                Direction := DirectionMode (*MC_DIRECTION*) , 
                                BufferMode := BufferMode (*MC_BUFFER_MODE*) ); 
        State := ERROR_STEP;
     END_IF ;
    ERROR_STEP:
     (* Adding code here, when motion got errors *)
    
    NEXT_STEP:
     (* Adding code here, after MC_Stop calling *)

     END_CASE ;
